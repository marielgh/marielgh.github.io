from nltk.tokenize import sent_tokenize, word_tokenize
import matplotlib.pyplot as plt

with open ('holy_grail.txt', "r") as myfile:
    holy_grail = myfile.read()

# Split the script into lines: lines
lines = re.split('\n',holy_grail)
print(lines[:5])

# Replace all script lines for speaker
pattern_prompts = "[A-Z]{2,}(?:\s|-)?(?:#\d)?(?:[A-Z]{2,})?(?:\s[A-Z]{2,})?:"
# [A-Z] means a character of the set {A,B,C,...,Z}
# ?: makes the group non-capuring
# ()? means that the group can appear 0 or more times
# {2,} means that a character of the set can appear 2 or more times

lines = [re.sub(pattern_prompts, '', l) for l in lines]
print(lines[:5])

# Tokenize each line: tokenized_lines
tokenized_lines = [regexp_tokenize(s,"\w+") for s in lines]

# Make a frequency list of lengths: line_num_words
line_num_words = [len(t_line) for t_line in tokenized_lines]

# Plot a histogram of the line lengths
plt.hist(line_num_words)
# Show the plot
plt.show()

from collections import Counter
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer

with open ("wiki_debugging.txt", "r") as myfile:
    article = myfile.read()

# Tokenize the article: tokens
tokens = regexp_tokenize(article, r'\b\w\w+\b')

# Convert the tokens into lowercase: lower_tokens
lower_tokens = [t.lower() for t in tokens]

# Remove all stop words: no_stops
no_stops = [t for t in lower_tokens if t not in stopwords.words('english')]

# Instantiate the WordNetLemmatizer
wordnet_lemmatizer = WordNetLemmatizer()

# Lemmatize all tokens into a new list: lemmatized
lemmatized = [wordnet_lemmatizer.lemmatize(t) for t in no_stops]

# Create the bag-of-words: bow
bow = Counter(lemmatized)

# Print the 10 most common tokens
print(bow.most_common(10))

import nltk 
from nltk.tokenize import sent_tokenize, word_tokenize
from collections import defaultdict
import numpy as np

with open ("newsarticles/uber_apple.txt", "r") as myfile:
    article = myfile.read()

sentences = sent_tokenize(article)
token_sentences = [word_tokenize(sent) for sent in sentences]

# Tag each tokenized sentence into parts of speech: pos_sentences
pos_sentences = [nltk.pos_tag(sent) for sent in token_sentences] # list of lists (of tuples)

# The ne_chunk function acts as a chunker, meaning it produces 2-level trees
# If binary=True, named entities are tagged as NE
# otherwise, the classifier adds category labels such as PERSON, ORGANIZATION, and GPE

for sent in pos_sentences[11:12]:
    print(nltk.ne_chunk(sent, binary=False),'\n')

# Also a built-in function to chunk sentences:
chunked_sentences = nltk.ne_chunk_sents(pos_sentences, binary=False)

# Look for stems of the tree with 'PERSON' labels
for sent in chunked_sentences:
    for chunk in sent:
        if hasattr(chunk, "label"):
            print(chunk,chunk.label())

# Chart with percentages for each category
chunked_sentences = nltk.ne_chunk_sents(pos_sentences, binary=False)
ner_categories = defaultdict(int)
            
for sent in chunked_sentences:
    for chunk in sent:
        if hasattr(chunk, 'label'):
            ner_categories[chunk.label()] += 1
            
labels = list(ner_categories.keys())
values = [ner_categories.get(l) for l in labels]

plt.pie(values, labels=labels, autopct='%1.1f%%', startangle=0)
plt.show()

import spacy
from polyglot.text import Text

with open ("newsarticles/uber_apple.txt", "r") as myfile:
    article = myfile.read()

nlp = spacy.load('en',tagger=False, parser=False, matcher=False)
doc = nlp(article)
spacy_ents = [ent.text for ent in doc.ents]
print('Spacy entities:\n',spacy_ents)

txt = Text(article)
polyglot_ents = [' '.join(ent) for ent in txt.entities]
print('\nPolyglot entities:\n',polyglot_ents)

ensemble_ents = set(spacy_ents).intersection(set(polyglot_ents))
print('\nEnsemble entities:\n',ensemble_ents)

num_left_out = len(set(spacy_ents).union(set(polyglot_ents))) - len(ensemble_ents)
print(num_left_out)
