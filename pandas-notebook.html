import pandas as pd
import matplotlib.pyplot as plt

weather = pd.read_csv('Austinweather.csv', index_col='Date', parse_dates=True)
print(weather.head(),'\n')
print(weather.info())

plt.style.use('ggplot')
weather.plot(subplots=True,figsize=(16,10))
plt.show()

# Rolling mean
unsmoothed = weather['Temperature']
smoothed = unsmoothed.rolling(window=24).mean()
weather['Temperature'].plot(alpha=0.4)
smoothed.plot(figsize=(16,3),linewidth=2,label='moving average',legend=True)
plt.title('Moving average of temperature over time (24h window)')
plt.show()

print(weather.loc['2010-01-01']) # print all entries corresponding to a certain date

# Downsampling
fig, axes = plt.subplots(nrows=1, ncols=2)

# Downsample to obtain only the daily highest temperatures in August: august_highs
august = weather['Temperature'].loc['2010 August']
august_highs = august.resample('D').max()
august_highs.plot(ax=axes[0],figsize=(16,5))
axes[0].set_ylabel('Highest temperature')

# Downsample to obtain the daily lowest temperatures in February: february_lows
february = weather['Temperature'].loc['2010 February']
february_lows = february.resample('D').min()
february_lows.plot(ax = axes[1])
axes[1].set_ylabel('Lowest temperature')
plt.show()

import matplotlib.pyplot as plt

df = pd.read_csv('austin.csv',header=None)

column_labels = 'Wban,date,Time,StationType,sky_condition,sky_conditionFlag,visibility,visibilityFlag,wx_and_obst_to_vision,wx_and_obst_to_visionFlag,dry_bulb_faren,dry_bulb_farenFlag,dry_bulb_cel,dry_bulb_celFlag,wet_bulb_faren,wet_bulb_farenFlag,wet_bulb_cel,wet_bulb_celFlag,dew_point_faren,dew_point_farenFlag,dew_point_cel,dew_point_celFlag,relative_humidity,relative_humidityFlag,wind_speed,wind_speedFlag,wind_direction,wind_directionFlag,value_for_wind_character,value_for_wind_characterFlag,station_pressure,station_pressureFlag,pressure_tendency,pressure_tendencyFlag,presschange,presschangeFlag,sea_level_pressure,sea_level_pressureFlag,record_type,hourly_precip,hourly_precipFlag,altimeter,altimeterFlag,junk'
list_to_drop = ['sky_conditionFlag', 'visibilityFlag', 'wx_and_obst_to_vision', 'wx_and_obst_to_visionFlag', 'dry_bulb_farenFlag', 'dry_bulb_celFlag', 'wet_bulb_farenFlag', 'wet_bulb_celFlag', 'dew_point_farenFlag', 'dew_point_celFlag', 'relative_humidityFlag', 'wind_speedFlag', 'wind_directionFlag', 'value_for_wind_character', 'value_for_wind_characterFlag', 'station_pressureFlag', 'pressure_tendencyFlag', 'pressure_tendency', 'presschange', 'presschangeFlag', 'sea_level_pressureFlag', 'hourly_precip', 'hourly_precipFlag', 'altimeter', 'record_type', 'altimeterFlag', 'junk']

column_labels_list = column_labels.split(',')
df.columns = column_labels_list
df_dropped = df.drop(list_to_drop,axis='columns')

# Convert the date column to string: df_dropped['date']
df_dropped['date'] = df_dropped['date'].astype(str)
# Pad leading zeros to the Time column: df_dropped['Time']
df_dropped['Time'] = df_dropped['Time'].apply(lambda x:'{:0>4}'.format(x))

# Concatenate the new date and Time columns: date_string
date_string = df_dropped['date'] + df_dropped['Time']
# Convert the date_string Series to datetime: date_times
date_times = pd.to_datetime(date_string, format='%Y%m%d%H%M')
# Set the index to be the new date_times container: df_clean
df_clean = df_dropped.set_index(date_times)

print(df_clean.info())

# Convert some columns to numeric values forcing the 'M' values (missing) to be translated into NaNs
g = lambda x : pd.to_numeric(df_clean[x], errors='coerce')
df_clean['dry_bulb_faren'] = g('dry_bulb_faren')
df_clean['wind_speed'] = g('wind_speed')
df_clean['dew_point_faren'] = g('dew_point_faren')
df_clean['visibility'] = g('visibility')
print(df_clean.loc['2011 June 20 8:00':'2011 June 20 9:00','dry_bulb_faren'])

# selecting certain sky conditions
sunny = df_clean.loc[df_clean['sky_condition']=='CLR']  # equals 
overcast = df_clean.loc[df_clean['sky_condition'].str.contains('OVC')]  # contains

# Resample sunny and overcast, aggregating by maximum daily temperature
sunny_daily_max = sunny.resample('D').dry_bulb_faren.max()
overcast_daily_max = overcast.resample('D').dry_bulb_faren.max()

# Weekly mean of visibility and dry_bulb_faren and correlation between the two
weekly_mean = df_clean[['visibility','dry_bulb_faren']].resample('W').mean()
print(weekly_mean.corr())  # print correlation coeff of visibility and dry_bulb_faren
# Plot weekly_mean with subplots=True
weekly_mean.plot(subplots=True,figsize=(13,7))
plt.show()

# sunny fraction boxplot
sunny = df_clean['sky_condition'] == 'CLR'
sunny_hours = sunny.resample('D').sum()  # resample by day and sum 
total_hours = sunny.resample('D').count()  # resample by day and count
sunny_fraction = sunny_hours / total_hours
